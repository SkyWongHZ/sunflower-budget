# 记账本接口文档

> 除登录接口外，所有接口都需要在请求头中携带 token
```javascript
Headers: {
  "Authorization": "Bearer ${token}"  // 登录接口返回的 token
}
```

## 用户模块

### 用户登录
```javascript
POST /api/auth/login

请求参数:
{
  "username": string, // 用户名
  "password": string  // 密码
}

返回数据:
{
  "code": 200,        // 状态码
  "msg": "success",   // 提示信息
  "data": {
    "token": string,  // 登录令牌
    "user": {
      "id": string,
      "username": string
    }
  }
}
```

## 标签模块

### 查询标签列表
```javascript
GET /api/tags

请求参数(Query参数):
{
  "type": string,      // 可选 标签类型
  "pageIndex": number, // 可选 默认1
  "pageSize": number   // 可选 默认20
}

返回数据:
{
  "code": 200,         // 状态码
  "msg": "success",    // 提示信息
  "data": {
    "total": number,   // 总条数
    "list": [{
      "id": string,    // 标签唯一id
      "name": string,  // 标签名称
      "icon": string,  // 标签图标
      "type": string   // 标签类型,支出和收入
    }]
  }
}
```

### 新增标签
```javascript
POST /api/tags

请求参数:
{
  "name": string,    // 标签名称
  "icon": string,    // 标签icon
  "type": string     // 标签类型,支出和收入
}

返回数据:
{
  "code": 200,       // 状态码
  "msg": "success",  // 提示信息
  "data": {
    "id": string
  }
}
```

### 编辑标签
```javascript
PUT /api/tags/:id

请求参数:
{
  "name": string,    // 标签名称
  "icon": string,    // 标签icon
  "type": string     // 标签类型,支出和收入
}

返回数据:
{
  "code": 200,       // 状态码
  "msg": "success",  // 提示信息
  "data": {
    "id": string
  }
}
```

### 删除标签
```javascript
DELETE /api/tags/:id

返回数据:
{
  "code": 200,       // 状态码
  "msg": "success"   // 提示信息
}
```

## 账单模块

### 查询账单列表
```javascript
GET /api/records

请求参数(Query参数):
{
  "tagId": string,      // 标签id
  "pageIndex": number,  // 可选 默认1
  "pageSize": number,   // 可选 默认20
  "startDate": string,  // 可选 开始日期 YYYY-MM-DD
  "endDate": string,    // 可选 结束日期 YYYY-MM-DD
  "type": string        // 可选 'income' | 'expense'
}

返回数据:
{
  "code": 200,          // 状态码
  "msg": "success",     // 提示信息
  "data": {
    "total": number,    // 总条数
    "list": [{
      "id": string,     // 标签唯一id
      "type": string,   // 标签类型,支出和收入
      "recordTime": string,  // 记录时间 YYYY-MM-DD HH:mm:ss
      "remark": string,      // 备注
      "amount": number,      // 金额
      "tag": {               // 关联的标签信息
        "id": string,
        "name": string,
        "icon": string
      }
    }]
  }
}
```

### 新增账单
```javascript
POST /api/records

请求参数:
{
  "amount": number,     // 账单金额 必填
  "tagId": string,      // 标签id 必填
  "type": string,       // 标签类型,支出和收入 必填
  "remark": string,     // 可选 备注
  "recordTime": string  // 记录时间 必填 YYYY-MM-DD HH:mm:ss
}

返回数据:
{
  "code": 200,         // 状态码
  "msg": "success",    // 提示信息
  "data": {
    "id": string
  }
}
```

### 编辑账单
```javascript
PUT /api/records/:id

请求参数:
{
  "type": string,       // 可选
  "amount": number,     // 可选
  "remark": string,     // 可选
  "tagId": string,      // 可选
  "recordTime": string  // 可选 YYYY-MM-DD HH:mm:ss
}

返回数据:
{
  "code": 200,         // 状态码
  "msg": "success",    // 提示信息
  "data": {
    "id": string
  }
}
```

### 删除账单
```javascript
DELETE /api/records/:id

返回数据:
{
  "code": 200,        // 状态码
  "msg": "success"    // 提示信息
}
```


### 手动触发账单统计
```javascript
POST /api/statistic/trigger


请求参数:
{
  "type": string,       
  "date": string,    
}

返回数据:
{
  "code": 200,        // 状态码
  "msg": "success"    // 提示信息
  "data":{
    "message": "统计任务已添加到队列"
  }
}

```




  ### 获取账单统计
  ```javascript
  GET /api/statistic/getStatistic


  请求参数:
  {
    "type": string,

          "icon": string,   // 标签图标
          "type": string,   // 标签类型 income/expense
          "count": number,  // 该标签的记录数量
          "amount": number  // 该标签的总金额
        },
        "tagId2": {
          // ... 其他标签的统计
        }
        }  
        "updatedAt": string  
        "createdAt": string,    // 创建时间
    }
  }

  ```
